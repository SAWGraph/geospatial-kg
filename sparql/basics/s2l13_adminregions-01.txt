# S2L13_AdminRegions basics
# 
# Retrieve all county subdivisions and their associated S2 cells

PREFIX kwg-ont: <http://stko-kwg.geog.ucsb.edu/lod/ontology/>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>

SELECT * WHERE {
   	?s rdf:type kwg-ont:AdministrativeRegion_3 .
    ?o kwg-ont:sfOverlaps | kwg-ont:sfWithin ?s .
    ?o rdf:type kwg-ont:S2Cell_Level13 .
} ORDER BY ?s ?o
